<script>

    const { createApp, ref, watch, onMounted, onUnmounted } = Vue;

 

    createApp({

        setup() {

            // --- 狀態定義 ---

            const isMetronomeRunning = ref(false); // 節拍器是否運行中

            const bpm = ref(120); // 節拍速度 (BPM: Beats Per Minute)

            const showCPRModal = ref(false); // 控制 CPR 彈出視窗

            let metronomeInterval = null;

            let audioContext = null;

            let oscillator = null;

 

            // --- 聲音產生器 ---

            function setupAudio() {

                if (audioContext === null) {

                    audioContext = new (window.AudioContext || window.webkitAudioContext)();

                }

            }

 

            function playClick() {

                setupAudio();

               

                // 創建一個震盪器

                const oscillator = audioContext.createOscillator();

                const gainNode = audioContext.createGain();

 

                oscillator.connect(gainNode);

                gainNode.connect(audioContext.destination);

 

                // 設定聲音屬性 (短促的滴答聲)

                oscillator.type = 'square'; // 方波

                oscillator.frequency.setValueAtTime(440, audioContext.currentTime); // 頻率 440 Hz

                gainNode.gain.setValueAtTime(0.5, audioContext.currentTime);

 

                // 設定聲音結束時間 (非常短促)

                gainNode.gain.exponentialRampToValueAtTime(0.001, audioContext.currentTime + 0.05);

 

                oscillator.start(audioContext.currentTime);

                oscillator.stop(audioContext.currentTime + 0.05); // 播放 50 毫秒

            }

 

            // --- 節拍器邏輯 ---

            function startMetronome() {

                if (metronomeInterval) {

                    clearInterval(metronomeInterval);

                }

                if (bpm.value < 60) bpm.value = 60; // 防止過低

                if (bpm.value > 180) bpm.value = 180; // 防止過高

 

                // 計算每個節拍的時間間隔 (毫秒)

                const intervalTime = (60 / bpm.value) * 1000;

 

                metronomeInterval = setInterval(() => {

                    playClick(); // 播放節奏聲

                }, intervalTime);

 

                isMetronomeRunning.value = true;

            }

 

            function stopMetronome() {

                if (metronomeInterval) {

                    clearInterval(metronomeInterval);

                    metronomeInterval = null;

                }

                isMetronomeRunning.value = false;

            }

 

            function toggleMetronome() {

                if (isMetronomeRunning.value) {

                    stopMetronome();

                } else {

                    startMetronome();

                }

            }

 

            // 監聽 BPM 變化，若節拍器運行中則重新啟動以應用新速度

            watch(bpm, (newBPM, oldBPM) => {

                if (isMetronomeRunning.value) {

                    startMetronome();

                }

            });

 

 

            // --- 緊急功能邏輯 ---

            function callEmergency() {

                // 這是模擬一鍵報案，實際在手機上會嘗試撥打電話

                const phoneNumber = '119';

                if (confirm(`您確定要撥打緊急電話 ${phoneNumber} 嗎？`)) {

                    // window.location.href = `tel:${phoneNumber}`; // 實際在手機瀏覽器中使用

                    alert(`正在模擬撥打 ${phoneNumber}...`);

                }

            }

           

            // --- 生命周期鉤子 (清理工作) ---

            onUnmounted(() => {

                stopMetronome();

                if (audioContext) {

                    audioContext.close();

                }

            });

 

 

            // --- 返回供模板使用的狀態和方法 ---

            return {

                isMetronomeRunning,

                bpm,

                showCPRModal,

                toggleMetronome,

                callEmergency,

            };

        }

    }).mount('#app');

</script>
